## 代码示例

### SkillCard 组件实现

<CodePreview lang="tsx">
{`import React from 'react';
import { Card, Rate, Tag, Space } from 'antd';
import { CodeOutlined, CheckOutlined } from '@ant-design/icons';

interface SkillCardProps {
  name: string;
  level: number;
  icon?: string;
  tags?: string[];
  description?: string;
}

const SkillCard: React.FC<SkillCardProps> = ({
  name,
  level,
  icon,
  tags = [],
  description
}) => {
  // 图标映射
  const getIcon = () => {
    switch (icon) {
      case 'react':
        return <i className="devicon-react-original colored"></i>;
      case 'typescript':
        return <i className="devicon-typescript-plain colored"></i>;
      default:
        return <CodeOutlined />;
    }
  };

  return (
    <Card className="skill-card" size="small">
      <div className="skill-header">
        <div className="skill-icon">
          {getIcon()}
        </div>
        <div className="skill-info">
          <div className="skill-name">{name}</div>
          <Rate disabled defaultValue={level} count={5} />
        </div>
      </div>

      {description && (
        <div className="skill-description">
          {description}
        </div>
      )}

      {tags.length > 0 && (
        <div className="skill-tags">
          <Space size={[0, 8]} wrap>
            {tags.map(tag => (
              <Tag key={tag} color="blue" icon={<CheckOutlined />}>
                {tag}
              </Tag>
            ))}
          </Space>
        </div>
      )}
    </Card>
  );
};

export default SkillCard;`}
</CodePreview>
